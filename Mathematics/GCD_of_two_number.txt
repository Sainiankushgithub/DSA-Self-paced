class Solution
{
	public:
    int gcd(int A, int B) 
	{ 
	    int min_num=min(A,B);
	    for(int i=min_num;i>=1;i--)
	    {
	        if(A%min_num==0 && B%min_num==0)
	        {
	            break;
	        }
	        min_num--;
	    }
	    return min_num;
	} 
};
