class Solution
{
	public:
		string is_palindrome(int n)
		{
		  //  string s1="Yes";
		  //  string s2="no";
		   int temp=n;
		   int sum=0;
		   int r;
		   while(temp>0)
		   {
		       r=temp%10;
		       sum=sum*10+r;
		       temp=temp/10;
		   }
		   if(sum==n)
		   {
		       return "Yes";
		   }
		   else
		   return "No";
		}
};

//{ Driver Code Starts.
int main(){
    int T;
    cin >> T;
    while(T--)
    {
    	int n;
    	cin >> n;
    	Solution ob;
    	string ans = ob.is_palindrome(n);
    	cout << ans <<"\n";
    }
	return 0;
}
